a = [55,7,78,12,42]
def bubble_sort(a,N):
    l, r = 0, 1 #왼쪽 커서, 오른쪽 커서
    while r < N: #오른쪽 커서가 크기보다 작을 때
        if r - l == 1 and a[l] > a[r]: #왼쪽커서가 오른쪽보다 크면
            a[l], a[r] = a[r], a[l] #둘이 교체하고
            l, r = 0,1 #초기화
        if a[l] < a[r]: #정렬이 잘 되어있으면
            l += 1 #왼쪽커서이동
        r += 1
    return a
